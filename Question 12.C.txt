#include <stdio.h>
#include <math.h>

int main() {
    float principal, rate, time, ci;

    // Input
    printf("Enter Principal: ");
    scanf("%f", &principal);
    printf("Enter Rate of Interest: ");
    scanf("%f", &rate);
    printf("Enter Time (in years): ");
    scanf("%f", &time);

    // Calculation
    ci = principal * pow((1 + rate / 100), time) - principal;

    // Output
    printf("Compound Interest = %.2f\n", ci);

    return 0;
}
